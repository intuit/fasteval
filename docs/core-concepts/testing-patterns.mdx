# Testing Patterns

fasteval works with pytest's assertion system out of the box. You can use regular `assert` statements right alongside fasteval's decorator-based evaluation — nothing special required.

## Working with EvalResult

`fe.score()` returns an `EvalResult` object with everything you need to know about how the evaluation went.

![EvalResult Structure](docs/assets/evalresult-structure.svg)

```python
import fasteval as fe

@fe.correctness(threshold=0.8)
def test_qa():
    response = agent("What is 2+2?")
    result = fe.score(response, "4", input="What is 2+2?")

    # EvalResult properties:
    # result.passed          - True if all metrics passed their thresholds
    # result.aggregate_score - Weighted average of all metric scores (0.0-1.0)
    # result.metric_results  - List of individual MetricResult objects
    # result.eval_input      - The input data used for evaluation
```

Each `MetricResult` in `result.metric_results` gives you the breakdown per metric:

```python
# MetricResult properties:
# mr.metric_name  - Name of the metric (e.g., "correctness")
# mr.score        - The score (0.0-1.0)
# mr.passed       - Whether this metric passed its threshold
# mr.threshold    - The threshold used
# mr.reasoning    - LLM's explanation for the score
```

## Custom Assertions

You can use the returned `EvalResult` for more granular assertions beyond the basic threshold check.

### Assert on Aggregate Score

```python
@fe.correctness(threshold=0.7)
@fe.relevance(threshold=0.7)
def test_high_quality_response():
    response = agent(query)
    result = fe.score(response, expected)

    # Require higher aggregate score for critical tests
    assert result.aggregate_score > 0.85, \
        f"Quality too low: {result.aggregate_score:.2f}"
```

### Assert on Specific Metrics

```python
@fe.correctness(threshold=0.7)
@fe.relevance(threshold=0.7)
@fe.toxicity(threshold=0.9)
def test_with_metric_checks():
    response = agent(query)
    result = fe.score(response, expected)

    # Find and check specific metric
    for mr in result.metric_results:
        if mr.metric_name == "correctness":
            assert mr.score > 0.8, f"Correctness too low: {mr.score}"
        if mr.metric_name == "toxicity":
            assert mr.score > 0.95, "Response may contain harmful content"
```

### Assert with Custom Messages

```python
@fe.correctness(threshold=0.8)
def test_with_detailed_failure():
    response = agent(query)
    result = fe.score(response, expected)

    # Build detailed assertion message
    if not result.passed:
        failed = [mr for mr in result.metric_results if not mr.passed]
        details = "\n".join(
            f"  - {mr.metric_name}: {mr.score:.2f} (needed {mr.threshold})"
            for mr in failed
        )
        assert False, f"Evaluation failed:\n{details}"
```

## Combining with Traditional Assertions

Normal pytest assertions work fine next to fasteval decorators. We'd recommend using them to validate preconditions, check response format, or enforce business logic before the LLM eval runs:

### Validate Before Evaluating

```python
@fe.correctness(threshold=0.8)
@fe.toxicity(threshold=0.95)
def test_response_validation():
    response = agent(query)

    # Traditional assertions first
    assert response is not None, "Agent returned None"
    assert len(response) > 0, "Response is empty"
    assert len(response) < 10000, "Response too long"

    # Then run LLM evaluation
    fe.score(response, expected)
```

### Check Response Structure

```python
@fe.correctness(threshold=0.8)
def test_json_response():
    response = agent("Generate user profile as JSON")

    # Validate JSON structure
    import json
    data = json.loads(response)  # Raises if invalid JSON
    assert "name" in data, "Missing 'name' field"
    assert "email" in data, "Missing 'email' field"

    # Then evaluate content quality
    fe.score(response, expected_profile)
```

### Business Logic Validation

```python
@fe.correctness(threshold=0.8)
def test_price_calculation():
    response = agent("What's the total for 3 items at $10 each?")

    # Check business logic
    assert "$30" in response or "30 dollars" in response.lower(), \
        "Incorrect price calculation"

    # Evaluate explanation quality
    fe.score(response, "The total is $30")
```

## Error Handling Patterns

### Using EvaluationFailedError

When any metric fails its threshold, `fe.score()` raises `EvaluationFailedError`. You can catch this if you need custom handling:

```python
from fasteval.models import EvaluationFailedError

@fe.correctness(threshold=0.8)
def test_with_error_handling():
    response = agent(query)

    try:
        fe.score(response, expected)
    except EvaluationFailedError as e:
        # Access the full result
        print(f"Aggregate score: {e.result.aggregate_score:.2f}")

        # Log failed metrics
        for mr in e.result.metric_results:
            if not mr.passed:
                print(f"{mr.metric_name}: {mr.score:.2f}")
                print(f"  Reason: {mr.reasoning}")

        # Re-raise for pytest
        raise
```

### Conditional Failure Handling

```python
import os
from fasteval.models import EvaluationFailedError

@fe.correctness(threshold=0.8)
def test_with_conditional_handling():
    response = agent(query)

    try:
        fe.score(response, expected)
    except EvaluationFailedError as e:
        # In CI, fail immediately
        if os.getenv("CI"):
            raise

        # Locally, log and continue if close to threshold
        if e.result.aggregate_score > 0.7:
            print(f"Warning: Score {e.result.aggregate_score:.2f} below threshold")
        else:
            raise
```

## Common Patterns

### Multiple Assertions Per Test

```python
@fe.correctness(threshold=0.8)
def test_multi_step_validation():
    # Step 1: Get response
    response = agent(query)
    assert response, "No response received"

    # Step 2: Validate format
    assert not response.startswith("Error"), "Agent returned error"

    # Step 3: Run evaluation
    result = fe.score(response, expected)

    # Step 4: Additional checks on result
    assert result.aggregate_score > 0.75
```

### Reusable Assertion Helpers

```python
def assert_quality_score(result, min_score=0.8, metric_name=None):
    """Helper to assert on evaluation quality."""
    if metric_name:
        for mr in result.metric_results:
            if mr.metric_name == metric_name:
                assert mr.score >= min_score, \
                    f"{metric_name} score {mr.score:.2f} below {min_score}"
                return
        raise AssertionError(f"Metric {metric_name} not found")
    else:
        assert result.aggregate_score >= min_score, \
            f"Aggregate score {result.aggregate_score:.2f} below {min_score}"


@fe.correctness(threshold=0.7)
@fe.relevance(threshold=0.7)
def test_with_helper():
    response = agent(query)
    result = fe.score(response, expected)

    # Use helper for additional checks
    assert_quality_score(result, min_score=0.85)
    assert_quality_score(result, min_score=0.9, metric_name="correctness")
```

### Parametrized Tests with Assertions

```python
import pytest

@pytest.mark.parametrize("query,expected,min_score", [
    ("What is 2+2?", "4", 0.9),
    ("Capital of France?", "Paris", 0.85),
    ("Explain quantum physics", "quantum mechanics", 0.7),  # Lower bar for complex topics
])
@fe.correctness(threshold=0.7)
def test_parametrized_with_thresholds(query, expected, min_score):
    response = agent(query)
    result = fe.score(response, expected, input=query)

    assert result.aggregate_score >= min_score, \
        f"Score {result.aggregate_score:.2f} below required {min_score} for: {query}"
```

### Soft Assertions (Collect All Failures)

```python
@fe.correctness(threshold=0.8)
def test_multiple_queries():
    failures = []

    for query, expected in test_cases:
        response = agent(query)
        try:
            result = fe.score(response, expected, input=query)
            if result.aggregate_score < 0.85:
                failures.append(f"{query}: score {result.aggregate_score:.2f}")
        except Exception as e:
            failures.append(f"{query}: {e}")

    assert not failures, f"Failures:\n" + "\n".join(failures)
```

## Mocking LangGraph Nodes

When you're testing LangGraph agents, you'll often want to mock out specific nodes so you can isolate the behavior you actually care about. Two approaches work well here: direct patching and the factory pattern.

![LangGraph Mocking Strategies](docs/assets/langgraph-mocking-strategies.svg)

**Direct patching** uses Python's `unittest.mock` to swap out node functions at test time.

### How Direct Patching Works

Say you've got a LangGraph agent with a retrieval node:

```python
# my_agent/nodes.py
def retrieval_node(state):
    """Fetches documents from vector store."""
    docs = vector_store.search(state["query"])
    return {"context": docs}

# my_agent/graph.py
from langgraph.graph import StateGraph
from my_agent.nodes import retrieval_node

graph = StateGraph(AgentState)
graph.add_node("retrieve", retrieval_node)
graph.add_node("generate", generation_node)
graph.add_edge("retrieve", "generate")
agent = graph.compile()
```

To mock the retrieval node, patch it **where it's used** (in the graph module), not where it's defined:

```python
from unittest.mock import patch
import fasteval as fe

@patch("my_agent.graph.retrieval_node")
@fe.faithfulness(threshold=0.8)
def test_rag_with_mocked_retrieval(mock_retrieval):
    # Configure the mock to return controlled context
    mock_retrieval.return_value = {
        "context": ["Paris is the capital of France."]
    }

    result = agent.invoke({"query": "What is the capital of France?"})

    # Evaluate with the mocked context
    fe.score(
        result["output"],
        expected="Paris",
        context=["Paris is the capital of France."]
    )
```

### Using pytest-mock

If you prefer a cleaner API, `pytest-mock` gives you a `mocker` fixture:

```python
import fasteval as fe

@fe.faithfulness(threshold=0.8)
def test_with_mocker(mocker):
    # Patch using mocker fixture
    mock_retrieval = mocker.patch("my_agent.graph.retrieval_node")
    mock_retrieval.return_value = {"context": ["Mocked context"]}

    result = agent.invoke({"query": "test query"})
    fe.score(result["output"], context=["Mocked context"])
```

### Common Gotchas

**Wrong patch path:** Always patch where the function is imported, not where it's defined. This one bites everyone at least once.

```python
# Wrong - patches the original module
@patch("my_agent.nodes.retrieval_node")

# Correct - patches where the graph imports it
@patch("my_agent.graph.retrieval_node")
```

**Forgetting return value:** Mocks return `MagicMock` by default, which isn't what your node expects. Always set `return_value`:

```python
mock_retrieval.return_value = {"context": ["actual docs"]}
```

## Test Graph Factory

For more explicit control, we'd recommend structuring your agent as a factory that takes swappable nodes. It's a bit more verbose, but you get better test isolation and it won't break when you refactor.

### Creating a Factory

```python
# my_agent/graph.py
from langgraph.graph import StateGraph
from typing import Callable, Optional

def create_agent(
    retrieval_fn: Optional[Callable] = None,
    generation_fn: Optional[Callable] = None,
):
    """
    Factory function that creates an agent with injectable nodes.

    Args:
        retrieval_fn: Custom retrieval node (uses default if None)
        generation_fn: Custom generation node (uses default if None)

    Returns:
        Compiled LangGraph agent
    """
    graph = StateGraph(AgentState)

    # Use provided functions or defaults
    graph.add_node("retrieve", retrieval_fn or default_retrieval_node)
    graph.add_node("generate", generation_fn or default_generation_node)

    graph.add_edge("retrieve", "generate")
    graph.set_entry_point("retrieve")
    graph.set_finish_point("generate")

    return graph.compile()

# Production usage
agent = create_agent()
```

### Testing with the Factory

```python
import fasteval as fe
from my_agent.graph import create_agent

def mock_retrieval_node(state):
    """Mock that returns controlled context."""
    return {"context": ["Paris is the capital of France."]}

@fe.faithfulness(threshold=0.8)
def test_generation_with_mocked_retrieval():
    # Create agent with mocked retrieval
    test_agent = create_agent(retrieval_fn=mock_retrieval_node)

    result = test_agent.invoke({"query": "Capital of France?"})

    fe.score(
        result["output"],
        context=["Paris is the capital of France."]
    )
```

### Pytest Fixtures for Reusable Test Agents

```python
import pytest
import fasteval as fe
from my_agent.graph import create_agent

@pytest.fixture
def mock_retrieval_node():
    """Fixture providing a mock retrieval node."""
    def _retrieval(state):
        return {"context": ["Mocked context for testing"]}
    return _retrieval

@pytest.fixture
def test_agent(mock_retrieval_node):
    """Fixture providing an agent with mocked retrieval."""
    return create_agent(retrieval_fn=mock_retrieval_node)

@fe.faithfulness(threshold=0.8)
def test_agent_generation(test_agent):
    result = test_agent.invoke({"query": "test"})
    fe.score(result["output"], context=["Mocked context for testing"])
```

### Configurable Mock Fixtures

When different tests need different mock behaviors:

```python
import pytest
from my_agent.graph import create_agent

@pytest.fixture
def agent_factory():
    """Factory fixture for creating agents with custom mocks."""
    def _create(context_docs=None):
        def mock_retrieval(state):
            return {"context": context_docs or ["default context"]}
        return create_agent(retrieval_fn=mock_retrieval)
    return _create

@fe.faithfulness(threshold=0.8)
def test_with_specific_context(agent_factory):
    agent = agent_factory(context_docs=["Custom test context"])
    result = agent.invoke({"query": "test"})
    fe.score(result["output"], context=["Custom test context"])
```

## Mocking Tool Calls

When your agents use external tools, mock them so you can control behavior and test tool trajectory evaluation.

### Basic Tool Mocking

```python
from unittest.mock import patch
import fasteval as fe
from fasteval import ExpectedTool

@patch("my_agent.tools.search_database")
@fe.tool_call_accuracy(threshold=0.9)
def test_agent_searches_correctly(mock_search):
    # Configure mock tool response
    mock_search.return_value = {
        "results": [{"id": 1, "name": "John Doe"}]
    }

    response, tool_calls = agent.invoke("Find user John")

    # Verify correct tool was called with right intent
    expected = [ExpectedTool(name="search_database")]
    fe.score(response, tool_calls=tool_calls, expected_tools=expected)
```

### Testing Tool Arguments

```python
from unittest.mock import patch
import fasteval as fe
from fasteval import ExpectedTool

@patch("my_agent.tools.book_flight")
@fe.tool_args_match(threshold=0.9)
def test_booking_arguments(mock_book):
    mock_book.return_value = {"confirmation": "ABC123"}

    response, tool_calls = agent.invoke("Book a flight to NYC tomorrow")

    expected = [
        ExpectedTool(
            name="book_flight",
            args={"destination": "NYC"}  # Verify key arguments
        )
    ]
    fe.score(response, tool_calls=tool_calls, expected_tools=expected)
```

### Testing Tool Sequences

```python
from unittest.mock import patch
import fasteval as fe
from fasteval import ExpectedTool

@patch("my_agent.tools.search_flights")
@patch("my_agent.tools.book_flight")
@patch("my_agent.tools.send_confirmation")
@fe.tool_sequence(threshold=0.9)
def test_booking_workflow(mock_confirm, mock_book, mock_search):
    # Configure mocks (note: decorators are applied bottom-up)
    mock_search.return_value = {"flights": [{"id": "F1", "price": 200}]}
    mock_book.return_value = {"confirmation": "ABC123"}
    mock_confirm.return_value = {"sent": True}

    response, tool_calls = agent.invoke("Book the cheapest flight to NYC")

    # Verify tools called in correct order
    expected = [
        ExpectedTool(name="search_flights"),
        ExpectedTool(name="book_flight"),
        ExpectedTool(name="send_confirmation"),
    ]
    fe.score(response, tool_calls=tool_calls, expected_tools=expected)
```

### Combining Tool Mocking with RAG Evaluation

```python
from unittest.mock import patch
import fasteval as fe
from fasteval import ExpectedTool

@patch("my_agent.tools.search_knowledge_base")
@fe.tool_call_accuracy(threshold=0.9)
@fe.faithfulness(threshold=0.8)
def test_rag_agent_with_tools(mock_search):
    mock_context = ["Company policy: Returns accepted within 30 days."]
    mock_search.return_value = {"documents": mock_context}

    response, tool_calls = agent.invoke("What's the return policy?")

    expected_tools = [ExpectedTool(name="search_knowledge_base")]
    fe.score(
        response,
        tool_calls=tool_calls,
        expected_tools=expected_tools,
        context=mock_context
    )
```
