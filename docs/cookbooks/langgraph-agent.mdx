# Testing LangGraph Agents

This cookbook shows how to evaluate LangGraph agents using fasteval's tool trajectory metrics. We'll test that agents call the right tools, in the right order, with the right arguments.

## Sample LangGraph Agent

Here's a simple agent with tools that we'll test:

```python
# agent.py
from langgraph.graph import StateGraph, END
from langchain_openai import ChatOpenAI
from langchain_core.tools import tool
from typing import TypedDict, Annotated, List
import operator

# Define tools
@tool
def search_web(query: str) -> str:
    """Search the web for information."""
    # Simulated search
    return f"Search results for: {query}"

@tool
def get_weather(city: str) -> str:
    """Get current weather for a city."""
    # Simulated weather
    return f"Weather in {city}: 72°F, sunny"

@tool
def calculate(expression: str) -> str:
    """Evaluate a mathematical expression."""
    try:
        result = eval(expression)
        return f"Result: {result}"
    except:
        return "Error: Invalid expression"

# Agent state
class AgentState(TypedDict):
    messages: Annotated[List[dict], operator.add]
    tool_calls: Annotated[List[dict], operator.add]

# Create agent
def create_agent():
    llm = ChatOpenAI(model="gpt-4o-mini").bind_tools([search_web, get_weather, calculate])
    
    def call_model(state):
        response = llm.invoke(state["messages"])
        tool_calls = []
        
        if response.tool_calls:
            for tc in response.tool_calls:
                tool_calls.append({
                    "name": tc["name"],
                    "arguments": tc["args"],
                })
        
        return {
            "messages": [response],
            "tool_calls": tool_calls,
        }
    
    def execute_tools(state):
        tool_map = {"search_web": search_web, "get_weather": get_weather, "calculate": calculate}
        results = []
        
        for tc in state["tool_calls"]:
            tool_fn = tool_map[tc["name"]]
            result = tool_fn.invoke(tc["arguments"])
            results.append({"role": "tool", "content": result, "tool_call_id": tc["name"]})
        
        return {"messages": results, "tool_calls": []}
    
    def should_continue(state):
        last_message = state["messages"][-1]
        if hasattr(last_message, "tool_calls") and last_message.tool_calls:
            return "tools"
        return END
    
    graph = StateGraph(AgentState)
    graph.add_node("agent", call_model)
    graph.add_node("tools", execute_tools)
    graph.add_edge("tools", "agent")
    graph.add_conditional_edges("agent", should_continue, {"tools": "tools", END: END})
    graph.set_entry_point("agent")
    
    return graph.compile()

agent = create_agent()
```

## Testing Tool Call Accuracy

Verify the agent calls the correct tools for a given task.

```python
# test_agent_tools.py
import fasteval as fe
from fasteval.models import ExpectedTool, ToolCall
from agent import agent

def run_agent(query: str):
    """Run agent and extract tool calls."""
    result = agent.invoke({
        "messages": [{"role": "user", "content": query}],
        "tool_calls": [],
    })
    
    # Collect all tool calls from the run
    all_tool_calls = []
    for msg in result["messages"]:
        if hasattr(msg, "tool_calls") and msg.tool_calls:
            for tc in msg.tool_calls:
                all_tool_calls.append(ToolCall(
                    name=tc["name"],
                    arguments=tc["args"],
                ))
    
    # Get final response
    final_response = result["messages"][-1].content if result["messages"] else ""
    
    return final_response, all_tool_calls


@fe.tool_call_accuracy(threshold=0.9)
def test_weather_tool_called():
    """Test that weather queries use the get_weather tool."""
    query = "What's the weather like in Paris?"
    response, tool_calls = run_agent(query)
    
    expected_tools = [
        ExpectedTool(name="get_weather")
    ]
    
    fe.score(
        actual_output=response,
        tool_calls=tool_calls,
        expected_tools=expected_tools,
        input=query,
    )


@fe.tool_call_accuracy(threshold=0.9)
def test_search_tool_called():
```

## Testing Tool Sequence

Verify tools are called in the correct order for multi-step tasks.

```python
@fe.tool_sequence(threshold=0.9)
def test_multi_tool_sequence():
    """Test that agent calls tools in the right order for complex queries."""
    query = "First search for the population of Tokyo, then calculate what 10% of that would be"
    response, tool_calls = run_agent(query)
    
    # Should search first, then calculate
    expected_tools = [
        ExpectedTool(name="search_web"),
        ExpectedTool(name="calculate"),
    ]
    
    fe.score(
        actual_output=response,
        tool_calls=tool_calls,
        expected_tools=expected_tools,
        input=query,
    )


@fe.tool_sequence(threshold=0.8)
def test_research_then_summarize():
    """Test research workflow: search then process."""
    query = "Find information about electric cars and calculate the average range if models have 300, 350, and 400 mile ranges"
    response, tool_calls = run_agent(query)
    
    expected_tools = [
        ExpectedTool(name="search_web"),
        ExpectedTool(name="calculate"),
    ]
    
    fe.score(
        actual_output=response,
        tool_calls=tool_calls,
        expected_tools=expected_tools,
        input=query,
    )
```

## Combining Tool and Output Metrics

Test both tool usage AND response quality:

```python
@fe.tool_call_accuracy(threshold=0.9)
@fe.tool_args_match(threshold=0.8)
@fe.correctness(threshold=0.8)
def test_complete_agent_behavior():
    """Test tool usage and response quality together."""
    query = "What is 100 + 200 + 300?"
    response, tool_calls = run_agent(query)
    
    expected_tools = [
        ExpectedTool(name="calculate", args={"expression": "100 + 200 + 300"})
    ]
    
    fe.score(
        actual_output=response,
        expected_output="The result is 600",
        tool_calls=tool_calls,
        expected_tools=expected_tools,
        input=query,
    )
```

## Mocking Agent Nodes

For isolated testing, mock specific nodes:

### Direct Patching

```python
from unittest.mock import patch

@patch("agent.search_web")
@fe.tool_call_accuracy(threshold=0.9)
def test_with_mocked_search(mock_search):
    """Test agent with mocked search results."""
    mock_search.return_value = "Mocked search result: Python was created in 1991"
    
    query = "When was Python created?"
    response, tool_calls = run_agent(query)
    
    expected_tools = [ExpectedTool(name="search_web")]
    
    fe.score(
        actual_output=response,
        tool_calls=tool_calls,
        expected_tools=expected_tools,
        input=query,
    )
    
    mock_search.assert_called_once()
```

### Factory Pattern

```python
def create_test_agent(search_fn=None, weather_fn=None, calc_fn=None):
    """Create agent with injectable tool implementations."""
    # ... (inject provided functions or use defaults)
    pass

@pytest.fixture
def mock_agent():
    """Agent with all tools mocked."""

        search_fn=lambda q: "Mocked search",
        weather_fn=lambda c: "Mocked weather: 70°F",

@fe.tool_call_accuracy(threshold=0.9)
```

## Production Monitoring

Once your agent tests pass, monitor it in production with **fasteval-observe**:

```python
from fasteval_observe import observe
from fasteval_observe.sampling import AdaptiveSamplingStrategy
import fasteval as fe

@observe(
    sampling=AdaptiveSamplingStrategy(
        base_rate=0.02,         # 2% baseline
        error_rate=1.0,         # 100% on errors
        slow_threshold_ms=5000,
        slow_rate=0.5,          # 50% of slow calls
    ),
    run_evaluations=True,
    metadata={"agent_type": "langgraph"}
)
@fe.tool_call_accuracy(threshold=0.9)
@fe.tool_sequence(threshold=0.8, expected_sequence=["search_web", "calculate"])
async def production_agent(query: str):
    """Production agent with runtime monitoring."""
    agent = create_agent()
    result = await agent.ainvoke({"messages": [{"role": "user", "content": query}]})
    
    return result["messages"][-1].content

# Use in production - automatically monitored and evaluated
response = await production_agent("What's the weather in NYC and what's 10+5?")
```
