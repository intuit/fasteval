# Prompt Regression Testing

This cookbook shows how to prevent quality regressions when you're updating prompts, system instructions, or model configurations.

## Overview

## Golden Test Dataset

Start by putting together a set of test cases that capture the behavior you expect.

### Loading from CSV

```python
# golden_dataset.py
import csv
from pathlib import Path

def load_golden_dataset(path: str = "golden_test_cases.csv"):
    """Load golden dataset from CSV."""
    cases = []
    with open(path) as f:
        reader = csv.DictReader(f)
        for row in reader:
            cases.append({
                "id": row["id"],
                "category": row["category"],
                "input": row["input"],
                "expected": row["expected"],
                "min_score": float(row["min_score"]),
                "tags": row["tags"].split(",") if row["tags"] else [],
            })
    return cases

GOLDEN_TEST_CASES = load_golden_dataset()
```

### Evaluation Harness

```python
# eval_harness.py
import fasteval as fe
from langchain_openai import ChatOpenAI
from golden_dataset import GOLDEN_TEST_CASES

class PromptEvaluator:
    def __init__(self, model: str = "gpt-4o-mini"):
        self.model = model

    def evaluate_prompt(self, system_prompt: str, test_cases: list) -> dict:
        """Evaluate a prompt against test cases."""
        llm = ChatOpenAI(model=self.model)

        results = {
            "scores": [],
            "by_category": {},
            "failures": [],
        }

        for case in test_cases:
            # Generate response with this prompt
            response = llm.invoke([
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": case["input"]},
            ])

            # Evaluate
            score = self._evaluate_case(response.content, case)
            results["scores"].append(score)

            # Track by category
            cat = case["category"]
            if cat not in results["by_category"]:
                results["by_category"][cat] = []
            results["by_category"][cat].append(score)

            # Track failures
            if score < case["min_score"]:
                results["failures"].append({
                    "id": case["id"],
                    "score": score,
                    "expected_min": case["min_score"],
                })

        results["aggregate"] = sum(results["scores"]) / len(results["scores"])
        return results

    def _evaluate_case(self, response: str, case: dict) -> float:
        """Evaluate a single test case."""
        # Use fasteval for evaluation
        if "expected" in case:
            result = fe.evaluate(
                actual_output=response,
                expected_output=case["expected"],
                metrics=["correctness"],
            )
            return result.aggregate_score
        return 0.0
```

### Side-by-Side Comparison

```python
# compare_prompts.py
from eval_harness import PromptEvaluator
from prompts import PROMPTS
from golden_dataset import GOLDEN_TEST_CASES

def compare_prompts(baseline_version: str, candidate_version: str):
    """Compare two prompt versions."""
    evaluator = PromptEvaluator()

    print(f"Evaluating baseline ({baseline_version})...")
    baseline_results = evaluator.evaluate_prompt(
        PROMPTS[baseline_version],
        GOLDEN_TEST_CASES
    )

    print(f"Evaluating candidate ({candidate_version})...")
    candidate_results = evaluator.evaluate_prompt(
        PROMPTS[candidate_version],
        GOLDEN_TEST_CASES
    )

    # Compare
    delta = candidate_results["aggregate"] - baseline_results["aggregate"]

    print("\n=== COMPARISON RESULTS ===")
    print(f"Baseline ({baseline_version}): {baseline_results['aggregate']:.3f}")
    print(f"Candidate ({candidate_version}): {candidate_results['aggregate']:.3f}")
    print(f"Delta: {delta:+.3f}")

    # Category breakdown
    print("\nBy Category:")
    for cat in baseline_results["by_category"]:
        b_avg = sum(baseline_results["by_category"][cat]) / len(baseline_results["by_category"][cat])
        c_avg = sum(candidate_results["by_category"][cat]) / len(candidate_results["by_category"][cat])
        cat_delta = c_avg - b_avg
        print(f"  {cat}: {b_avg:.3f} → {c_avg:.3f} ({cat_delta:+.3f})")

    # Regressions
    if candidate_results["failures"]:
        print("\nNew Failures in Candidate:")
        for failure in candidate_results["failures"]:
            print(f"  - {failure['id']}: {failure['score']:.2f} (min: {failure['expected_min']})")

    return {
        "baseline": baseline_results,
        "candidate": candidate_results,
        "delta": delta,
        "passed": delta >= -0.05,  # Allow up to 5% regression
    }

if __name__ == "__main__":
    result = compare_prompts("v1", "v2")
    print(f"\nRegression Test: {'PASSED' if result['passed'] else 'FAILED'}")
```

## Automated Regression Detection

### pytest Integration

```python
# test_prompt_regression.py
import pytest
import fasteval as fe
from prompts import PROMPTS
from golden_dataset import GOLDEN_TEST_CASES, CATEGORIES

# Current production version
BASELINE_VERSION = "v1"
# Version being tested
CANDIDATE_VERSION = "v2"

# Acceptable regression threshold
MAX_REGRESSION = 0.05

@pytest.fixture(scope="module")
def baseline_scores():
    """Pre-computed baseline scores (or compute once)."""
    # In practice, load from a file or compute once
    return {"aggregate": 0.85, "by_category": {"factual_qa": 0.90, "reasoning": 0.82}}

@fe.correctness(threshold=0.7)
def test_candidate_absolute_quality():
    """Ensure candidate meets minimum quality bar."""
    llm = get_llm_with_prompt(PROMPTS[CANDIDATE_VERSION])

    for case in GOLDEN_TEST_CASES[:10]:  # Sample
        response = llm.invoke(case["input"])
        fe.score(response, expected=case["expected"], input=case["input"])

def test_no_regression_vs_baseline(baseline_scores):
    """Ensure candidate doesn't regress vs baseline."""
    candidate_scores = evaluate_prompt(PROMPTS[CANDIDATE_VERSION], GOLDEN_TEST_CASES)

    delta = candidate_scores["aggregate"] - baseline_scores["aggregate"]

    assert delta >= -MAX_REGRESSION, (
        f"Regression detected: {delta:.3f} "
        f"(baseline: {baseline_scores['aggregate']:.3f}, "
        f"candidate: {candidate_scores['aggregate']:.3f})"
    )

@pytest.mark.parametrize("category", ["factual_qa", "reasoning", "safety"])
def test_no_category_regression(category, baseline_scores):
    """Check each category for regression."""
    cases = CATEGORIES[category]
    candidate_scores = evaluate_prompt(PROMPTS[CANDIDATE_VERSION], cases)

    baseline_avg = baseline_scores["by_category"].get(category, 0.8)
    candidate_avg = candidate_scores["aggregate"]
    delta = candidate_avg - baseline_avg

    assert delta >= -MAX_REGRESSION, (
        f"Regression in {category}: {delta:.3f}"
    )
```

### Baseline Management

![Baseline Management Workflow](docs/assets/baseline-management.svg)

```python
# baseline_manager.py
import json
from datetime import datetime
from pathlib import Path

class BaselineManager:
    def __init__(self, baseline_dir: str = "baselines"):
        self.baseline_dir = Path(baseline_dir)
        self.baseline_dir.mkdir(exist_ok=True)

    def save_baseline(self, version: str, scores: dict):
        """Save evaluation scores as a baseline."""
        baseline = {
            "version": version,
            "timestamp": datetime.utcnow().isoformat(),
            "scores": scores,
        }

        path = self.baseline_dir / f"{version}.json"
        with open(path, "w") as f:
            json.dump(baseline, f, indent=2)

        print(f"Saved baseline for {version} to {path}")

    def load_baseline(self, version: str) -> dict:
        """Load a saved baseline."""
        path = self.baseline_dir / f"{version}.json"
        with open(path) as f:
            return json.load(f)

    def get_production_baseline(self) -> dict:
        """Get the current production baseline."""
        path = self.baseline_dir / "production.json"
        with open(path) as f:
            return json.load(f)

    def promote_to_production(self, version: str):
        """Promote a version's baseline to production."""
        baseline = self.load_baseline(version)

        # Archive current production
        try:
            current = self.get_production_baseline()
            archive_path = self.baseline_dir / f"production_{current['timestamp']}.json"
            with open(archive_path, "w") as f:
                json.dump(current, f, indent=2)
        except FileNotFoundError:
            pass

        # Set new production baseline
        path = self.baseline_dir / "production.json"
        with open(path, "w") as f:
            json.dump(baseline, f, indent=2)

        print(f"Promoted {version} to production baseline")
```

## CI Integration

### GitHub Action

```yaml
# .github/workflows/prompt-regression.yml
name: Prompt Regression Test

on:
  pull_request:
    paths:
      - 'prompts/**'
      - 'system_prompts.py'

jobs:
  regression-test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-python@v5
        with:
          python-version: '3.11'

      - name: Install dependencies
        run: pip install -e ".[test]"

      - name: Load baseline
        run: |
          # Download production baseline from artifact storage
          aws s3 cp s3://my-bucket/baselines/production.json baselines/

      - name: Run regression tests
        env:
          OPENAI_API_KEY: ${{ secrets.OPENAI_API_KEY }}
        run: pytest tests/test_prompt_regression.py -v

      - name: Generate comparison report
        if: always()
        run: python scripts/compare_prompts.py --output report.md

      - name: Comment on PR
        if: always()
        uses: actions/github-script@v7
        with:
          script: |
            const fs = require('fs');
            const report = fs.readFileSync('report.md', 'utf8');
            github.rest.issues.createComment({
              issue_number: context.issue.number,
              owner: context.repo.owner,
              repo: context.repo.repo,
              body: report
            });
```
